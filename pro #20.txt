float  CircleAreaInscribedInSquare(float A)
{
	float const pi = 3.141592653589793238;
	float Area = (pow(A, 2) * pi) / 4;
	return Area;
}
void PrintResult(float Area)
{
	cout << "\nCircle Area = " << Area << endl;
}
