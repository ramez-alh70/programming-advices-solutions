#include<iostream>
#include<cstring>
#include<string>

using namespace std;
float ReadNumbers(string Message )
{
	float Number = 0;

	cout << Message << endl;
	cin >> Number;
	return Number;


}
float SumNumbers()
{
	int Sum = 0;
	int counter = 1;
	int Number = 0;
	Number = ReadNumbers("Please enter The Number " + to_string(counter));
	while (Number!=-90)
	{
		Sum += Number;
		counter++;

		Number = ReadNumbers("Please enter Number " + to_string(counter));

	}
	return Sum;
}

int main()
{
	cout << endl << "Result is :" << SumNumbers();

	return 0;

}
